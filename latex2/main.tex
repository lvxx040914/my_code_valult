% ===================== 核心命令：单题一页，可自动跨页 =====================
\newcommand{\oneproblem}[2][0.42\textheight]{
    \clearpage
    \thispagestyle{fancy}
    \refstepcounter{problem}

    % ---------- 题目 ----------
    \begin{tcolorbox}[
        problemcard,
        overlay unbroken={
            \node[
                circle,
                fill=MainBlue,
                text=white,
                font=\bfseries,
                inner sep=1.5pt,
                minimum size=7mm
            ]
            at ([xshift=-1.5mm,yshift=-8mm]frame.north west) {\theproblem};
        },
        overlay first={
            \node[
                circle,
                fill=MainBlue,
                text=white,
                font=\bfseries,
                inner sep=1.5pt,
                minimum size=7mm
            ]
            at ([xshift=-1.5mm,yshift=-8mm]frame.north west) {\theproblem};
        }
    ]
        {\large #2}
    \end{tcolorbox}

    \vspace{0.65em}

    % 如果剩余空间太少，就把思路框放到下一页，避免只露一个标题
    \Needspace{0.18\textheight}

    % ---------- 思路与演算 ----------
    \begin{tcolorbox}[
        solutioncard,
        height=#1
    ]
        \vfill
    \end{tcolorbox}
}
