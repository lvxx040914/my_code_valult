\fancyhead[C]{
    \begin{tikzpicture}[remember picture, overlay]
        % 使用渐变色背景
        \fill[left color=MainBlue, right color=MainBlue!60] 
            ([yshift=-2.2cm]current page.north west) rectangle (current page.north east);
        
        % 底部白色装饰线条
        \draw[white, line width=1pt] ([yshift=-2.2cm]current page.north west) -- ([yshift=-2.2cm]current page.north east);
        
        % 左侧文字
        \node[anchor=west, white, font=\Large\bfseries] at ([xshift=1.5cm, yshift=-1.1cm]current page.north west) {
            \currentday \ \scalebox{0.8}{|} \ \currenttopic
        };
        
        % 右侧文字：包含 Page 字样
        \node[anchor=east, white!90, font=\small\itshape] at ([xshift=-1.5cm, yshift=-1.1cm]current page.north east) {
            100 Days Challenge \quad Page \thepage
        };
    \end{tikzpicture}
}
