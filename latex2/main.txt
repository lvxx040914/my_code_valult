% --- 核心：自动生成 PDF 书签与重置题号 ---
\newcommand{\setday}[2]{
    \newpage 
    \renewcommand{\currentday}{第 #1 天}
    \renewcommand{\currenttopic}{#2}
    \setcounter{probcount}{0} % 每天重置题号
    \phantomsection
    \addcontentsline{toc}{section}{第 #1 天：#2}
}

% --- 顶栏美化 (使用渐变色) ---
\pagestyle{fancy}
\fancyhf{} 
\renewcommand{\headrulewidth}{0pt}
\fancyhead[C]{
    \begin{tikzpicture}[remember picture, overlay]
        % 渐变色背景顶栏
        \fill[left color=MainBlue, right color=MainBlue!60] 
            ([yshift=-2.2cm]current page.north west) rectangle (current page.north east);
        % 底部装饰线条
        \draw[white, line width=1pt] ([yshift=-2.2cm]current page.north west) -- ([yshift=-2.2cm]current page.north east);
        
        \node[anchor=west, white, font=\Large\bfseries] at ([xshift=1.5cm, yshift=-1.1cm]current page.north west) {
            \currentday \ \scalebox{0.8}{|} \ \currenttopic
        };
        \node[anchor=east, white!90, font=\small\itshape] at ([xshift=-1.5cm, yshift=-1.1cm]current page.north east) {
            100 Days Challenge \quad Page \thepage
        };
    \end{tikzpicture}
}

% --- 题目框样式定义 ---
\newtcolorbox{problembox}[1]{
    colback=white,
    colframe=MainBlue,
    fonttitle=\bfseries\large,
    title=题目 #1,
    enhanced,
    attach boxed title to top left={yshift=-2mm, xshift=5mm},
    boxed title style={colback=MainBlue, sharp corners, boxrule=0pt},
    sharp corners,
    boxrule=0.8pt,
    top=12pt,
    bottom=10pt,
    left=10pt,
    right=10pt,
    before skip=20pt, % 题目之间的间距
    after skip=10pt,
    breakable,        % 【关键】允许题目框自动跨页断开
}

% --- 核心：一键出题命令 ---
\newcommand{\oneproblem}[1]{
    \newpage 
    \stepcounter{probcount} % 题号自增
    \begin{problembox}{\arabic{probcount}}
        {\large #1}
    \end{problembox}
    \vfill % 撑开页面，留出解题空间
}
